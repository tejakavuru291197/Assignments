using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace pawan
{
    internal class Average
    {
       static void Main(string[] args)
        {
            int[] arr = new int[5];
            int max = arr[0];
            int total = 0;
            for (int i = 0; i < arr.Length; i++)
            {
                Console.WriteLine("Enter " + i + " Student Marks:");
                arr[i] = int.Parse(Console.ReadLine());
                total = total + arr[i];
            }



            for (int i = 0; i < arr.Length; i++)
            {
                if (arr[i] > max)
                    max = arr[i];
            }
            int avg = total / arr.Length;
            Console.WriteLine("Avg Of Students Marks {0}", avg);
            Console.WriteLine("highest marks is {0}", max);




            Console.ReadLine();







        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace pawan
{
    internal class ParamArray
    {
      public static int sum(int[] arrr)
        {
            int total = 0;
            for (int i = 0; i < arrr.Length; i++)
            {
                total += arrr[i];
            }
            return total;
        }
        public static void Main()
        {
            int[] arr = new int[5] { 5, 9, 7, 2, 3 };
            Console.WriteLine("total sum of array is {0}", sum(arr));
            Console.ReadLine();

        }
    }

}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace pawan
{
    internal class Swapping
    {
        public static void Swap()
        {
            int a = 5, b = 6, temp;
            Console.WriteLine("Before Swapping a= {0}, b= {1}", a, b);
            temp = a;
            a = b;
            b = temp;
            Console.WriteLine("After Swapping a= {0}, b= {1}", a, b);
        }
        public static void Main()
        {
            Swap();
            Console.ReadLine();
        }
    }
}
